# rules for an agent to fetch cubes in the environment.

# s3 and s4 are cube sensors on the right and left of agent's heading.
# if activation from cubes is higher on the left, then go towards left.
if s3 < s4 {
   a0 = 1
   a1 = 1
}
# otherwise, go towards right (notice that this action is also taken
# if both sensor values are the same, i.e. the agent will go around in
# a clockwise circle until s3/s4 detects cubes).
else {
   a0 = 1
   a1 = -1
}
